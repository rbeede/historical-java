64  package org.apache.commons.httpclient;
65  
66  /***
67   * <p>Username and password {@link Credentials}.</p>
68   *
69   * @author <a href="mailto:remm@apache.org">Remy Maucherat</a>
70   * @author Sean C. Sullivan
71   * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
72   * 
73   * @version $Revision: 1.11 $ $Date: 2003/01/31 00:33:36 $
74   * 
75   */
76  public class UsernamePasswordCredentials implements Credentials {
77  
78      // ----------------------------------------------------------- Constructors
79  
80      /***
81       * Default constructor.
82       */
83      public UsernamePasswordCredentials() {
84      }
85  
86  
87      /***
88       * The constructor with the username and password combined string argument.
89       *
90       * @param usernamePassword the username:password formed string
91       * @see #toString
92       */
93      public UsernamePasswordCredentials(String usernamePassword) {
94          int atColon = usernamePassword.indexOf(':');
95          if (atColon >= 0) {
96              this.userName = usernamePassword.substring(0, atColon);
97              this.password = usernamePassword.substring(atColon + 1);
98          } else {
99              this.userName = usernamePassword;
100         }
101     }
102 
103 
104     /***
105      * The constructor with the username and password arguments.
106      *
107      * @param userName the user name
108      * @param password the password
109      */
110     public UsernamePasswordCredentials(String userName, String password) {
111         this.userName = userName;
112         this.password = password;
113     }
114 
115     // ----------------------------------------------------- Instance Variables
116 
117     /***
118      * User name.
119      */
120     private String userName;
121 
122 
123     /***
124      * Password.
125      */
126     private String password;
127 
128 
129     // ------------------------------------------------------------- Properties
130 
131 
132     /***
133      * User name property setter.
134      *
135      * @param userName
136      * @see #getUserName()
137      */
138     public void setUserName(String userName) {
139         this.userName = userName;
140     }
141 
142 
143     /***
144      * User name property getter.
145      *
146      * @return the userName
147      * @see #setUserName(String)
148      */
149     public String getUserName() {
150         return userName;
151     }
152 
153 
154     /***
155      * Password property setter.
156      *
157      * @param password
158      * @see #getPassword()
159      */
160     public void setPassword(String password) {
161         this.password = password;
162     }
163 
164 
165     /***
166      * Password property getter.
167      *
168      * @return the password
169      * @see #setPassword(String)
170      */
171     public String getPassword() {
172         return password;
173     }
174 
175     
176     /***
177      * Get this object string.
178      *
179      * @return the username:password formed string
180      */
181     public String toString() {
182         StringBuffer result = new StringBuffer();
183         result.append((this.userName == null) ? "null" : this.userName);
184         result.append(":");
185         result.append((this.password == null) ? "null" : this.password);
186         return result.toString();
187     }
188 
189 }
